Unit: DeLaFitsCommon


Type: function


function BitPixToInt(Value: TBitPix): Integer;


Description:


The function converts TBitPix into Integer, by the standard. The result - the number of bits of TBitPix.

For floating-point numbers (bi64f and bi32f) function value is less than 0.

For the value biUnknown function returns 0.


Code Sample:


var

 R64f, R32f: Integer;

 R08u, R16c, R32c, R64c: Integer;

 Rxxx: Integer;

begin

 R64f := BitPixToInt(bi64f);

 R32f := BitPixToInt(bi32f);

 R08u := BitPixToInt(bi08u);

 R16c := BitPixToInt(bi16c);

 R32c := BitPixToInt(bi32c);

 R64c := BitPixToInt(bi64c);

 Rxxx := BitPixToInt(biUnknown);

end;


Output


R64f = -64

R32f = -32

R08u = 8

R16c = 16

R32c = 32

R64c = 64

Rxxx = 0
