property ValuesRa[Index: Integer]: Extended Index cCastRa read GetValuesCoord

   write SetValuesCoord;


property ValuesDe[Index: Integer]: Extended Index cCastDe read GetValuesCoord

   write SetValuesCoord;


Description:


List of the Values from the lines of header as the coordinate: Right ascension and Declination.

Line − is a record from of items <keyword-separator-value-separator-note>, see TLineItem.


TLineBuilder.Count returns the number of significant lines of header (including END).


Numbering of lines a header begins with zero


Code Sample:


Input



10

11

header fragment

RA      = '22 27 58.07'        / [hms J2000] Target right ascension            

DEC     = '-11 40 26.8'        / [dms +N J2000] Target declination              


var

 Fit: TFitsHeader;

 Ra, De: Double;

begin

 // open file for read and write

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileReadWrite));

 // read items value as coordinates

 Ra := Fit.LineBuilder.ValuesRa[10];

 De := Fit.LineBuilder.ValuesDe[11];

 // write items value as coordinates

 Fit.LineBuilder.ValuesRa[10] := 15 + 7.5; // ~1.5h

 Fit.LineBuilder.ValuesDe[11] := 1.5;      // ~1.5°

 // free

 Fit.Stream.Free;

 Fit.Free;

end;


Output


// read items value as coordinates


Ra = 336.99195833

De = -11.674111111


// write items value as coordinates



10

11

header fragment

RA      = '01:30:00.00'        / [hms J2000] Target right ascension            

DEC     = '+01:30:00.0'        / [dms +N J2000] Target declination
