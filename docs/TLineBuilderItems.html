Type: property


Visibility: public


Member Of: TLineBuilder


property Items[Index, Cast: Integer]: TLineItem read GetItems write SetItems;


Description:


List of the lines of header.

Line - is a record from of items <keyword-separator-value-separator-note>, see TLineItem.


TLineBuilder.Count returns the number of significant lines of header (including END).


Numbering of lines a header begins with zero


Code Sample:


Input



10

11

header fragment

SOMEKEY1=                  123 / integer value                                  

SOMEKEY2=                      / null                                          


var

 Fit: TFitsHeader;

 Item1, Item2: TLineItem;

begin

 // open file for read and write

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileReadWrite));

 // get item

 Item1 := Fit.LineBuilder.Items[10, cCastInteger];

 // set item

 Item2.Keyword       := 'SOMEKEY2';

 Item2.ValueIndicate := True;

 Item2.Value.Ext     := 123.456;

 Item2.NoteIndicate  := True;

 Item2.Note          := 'new float value';

 Fit.LineBuilder.Items[11, cCastFloat] := Item2;

 // free

 Fit.Stream.Free;

 Fit.Free;

end;

Item1.ValueIndicate = True;

Item1.Value.Int     = 123;

Item1.NoteIndicate  = True;
