property GraphicRgn: TRgn read GetGraphicRgn;


Description:


Full region of the graphic image of the data.


Specifies coordinates and size of the full region after a series of Affine transformation. The coordinates of the full region are defined relative to of "zero scene" TFitsFrame.DataRgn.


Map of data: coordinate system

var

 Fit: TFitsGraphic;

 Rgn0, Rgn1, Rgn2: TRgn;

begin

 // open file for read

 Fit := TFitsGraphic.CreateJoin(TFileStream.Create('some.fit', cFileRead));


 // get full data region ≡ 'zero scene' of graphic image of the data

 Rgn0 := Fit.DataRgn;


 // get full region of the graphic image of the data

 // before the affine transformation

 Rgn1 := Fit.GraphicRgn;


 // get full region of the graphic image of the data

 // after the affine transformation

 Fit.GraphicGeom.Scl(2.0, 2.0, xy00);

 Rgn2 := Fit.GraphicRgn;


 // ... some code


 // free

 Fit.Stream.Free;

 Fit.Free;

end;
