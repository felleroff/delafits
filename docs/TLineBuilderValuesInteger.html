property ValuesInteger[Index: Integer]: Int64 read GetValuesInteger

   write SetValuesInteger;


Description:


List of the Values from the lines of header as Integer-value.

Line - is a record from of items <keyword-separator-value-separator-note>, see TLineItem.


TLineBuilder.Count returns the number of significant lines of header (including END).


Numbering of lines a header begins with zero


Code Sample:


Input



10

header fragment

SOMEKEY =                    1 / integer value                                  


var

 Fit: TFitsHeader;

 V: Integer;

begin

 // open file for read and write

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileReadWrite));

 // read and write item value as integer

 V := Fit.LineBuilder.ValuesInteger[10];

 V := V + 1;

 Fit.LineBuilder.ValuesInteger[10] := V;

 // free

 Fit.Stream.Free;

 Fit.Free;

end;


Output



10

header fragment

SOMEKEY =                    2 / integer value
