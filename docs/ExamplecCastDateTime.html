An example of using the constants cCastDateTime, cCastDate, cCastTime


The parameters of the parse of TDateTime are available in TLineBuilder.Fmt: PFormatLine.


1. We will read the line of header as TDateTime


Input



10

11

12

header fragment

DATETIME= '2011-09-06T07:28:35' / [ISO 8601] UTC date/time of exposure start    

DATE    = '06/09/11'           / [old format] UTC date of exposure start        

TIME    = '07:28:35'           / [old format] UTC time of exposure start        


var

 Fit: TFitsFileFrame;

 Index: Integer;

 Item: TLineItem;

 DateTime, DateOnly, TimeOnly: TDateTime;

 S1, S2, S3: string;

begin

 // open file for read

 Fit := TFitsFileFrame.CreateJoin('some.fit', cFileRead);

 // search and read for line with a keyword 'DATETIME'

 Index := Fit.LineBuilder.IndexOf('DATETIME');

 if Index >= 0 then

 begin

   Item     := Fit.LineBuilder.Items[Index, cCastDateTime];

   DateTime := Item.Value.Dtm;

 end;

 // search and read for line with a keyword 'DATE'

 Index := Fit.LineBuilder.IndexOf('DATE');

 if Index >= 0 then

 begin

   // default short date format - yymmdd

   // yymmdd: '06/09/11' -> '2006 09 11';

   // ddmmyy: '06/09/11' -> '2011 09 06';

   Fit.LineBuilder.Fmt^.vaDateTime.rFmtShortDate := ddmmyy;

   Item     := Fit.LineBuilder.Items[Index, cCastDate];

   DateOnly := Item.Value.Dtm;

 end;

 // search and read for line with a keyword 'TIME'

 Index := Fit.LineBuilder.IndexOf('TIME');

 if Index >= 0 then

 begin

   Item     := Fit.LineBuilder.Items[Index, cCastTime];

   TimeOnly := Item.Value.Dtm;

 end;

 // output

 DateTimeToString(S1, 'YYYY.MM.DD hh:mm:ss.zzz', DateTime);

 DateTimeToString(S2, 'YYYY.MM.DD hh:mm:ss.zzz', DateOnly);

 DateTimeToString(S3, 'YYYY.MM.DD hh:mm:ss.zzz', TimeOnly);

 // free

 Fit.Free;

end;


Output


S1 = '2011.09.06 07:28:35.000'

S2 = '2011.09.06 00:00:00.000'

S3 = '1899.12.30 07:28:35.000'


2. We will add the line of header as TDateTime


var

 Fit: TFitsFileFrame;

 Item: TLineItem;

begin

 // open file for read and write

 Fit := TFitsFileFrame.CreateJoin('some.fit', cFileReadWrite);

 // build and add item as DateTime

 Item.Keyword       := 'DATETIME';

 Item.ValueIndicate := True;

 Item.Value.Dtm     := EncodeDate(2011, 09, 06) + EncodeTime(07, 28, 35, 000);

 Item.NoteIndicate  := True;

 Item.Note          := 'this is my value of date and time';

 Fit.LineBuilder.Add(Item, cCastDateTime);

 // build and add item as Date

 Item.Keyword       := 'DATE';

 Item.ValueIndicate := True;

 Item.Value.Dtm     := EncodeDate(2011, 09, 06);

 Item.NoteIndicate  := True;

 Item.Note          := 'this is my value of date only';

 Fit.LineBuilder.Add(Item, cCastDate);

 // build and add item as Time

 Item.Keyword       := 'TIME';

 Item.ValueIndicate := True;

 Item.Value.Dtm     := EncodeTime(07, 28, 35, 000);

 Item.NoteIndicate  := True;

 Item.Note          := 'this is my value of time only';

 Fit.LineBuilder.Add(Item, cCastTime);

 // free

 Fit.Free;

end;
