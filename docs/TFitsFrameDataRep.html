property DataRep: TRep read FDataRep write SetDataRep;


Description:


Type of the elements of buffer of user for read/write data.


By Standard, size and bit depth of the data frame specifies the BITPIX keyword. However, the user can specify the type of elements of buffer for reading and writing data:


  • DataRep := rep80f ⇒ Data: array of array of Extended;


  • DataRep := rep64f ⇒ Data: array of array of Double;


  • DataRep := rep32f ⇒ Data: array of array of Single;


  • DataRep := rep08c ⇒ Data: array of array of ShortInt;


  • DataRep := rep08u ⇒ Data: array of array of Byte;


  • DataRep := rep16c ⇒ Data: array of array of SmallInt;


  • DataRep := rep16u ⇒ Data: array of array of Word;


  • DataRep := rep32c ⇒ Data: array of array of LongInt;


  • DataRep := rep32u ⇒ Data: array of array of LongWord;


  • DataRep := rep64c ⇒ Data: array of array of Int64;

var

 Fit: TFitsFrame;

 Rgn: TRgn;

 RepAuto: TRep;

 Data64f: array of array of Double;

 Data16c: array of array of SmallInt;

 Data32c: array of array of Integer;

begin

 // open file for read

 Fit := TFitsFrame.CreateJoin(TFileStream.Create('some.fit', cFileRead));

 // define region

 Rgn.X1 := 0;

 Rgn.Y1 := 0;

 Rgn.ColsCount := 4;

 Rgn.RowsCount := 3;

 // get auto representation of data

 RepAuto := Fit.DataRep;

 // read a block of data into array of array of Double

 Fit.DataRep := rep64f;

 Fit.DataPrepare(Pointer(Data64f), Rgn);

 Fit.DataRead(Pointer(Data64f), Rgn);

 // read a block of data into array of array of SmallInt

 Fit.DataRep := rep16c;

 Fit.DataPrepare(Pointer(Data16c), Rgn);

 Fit.DataRead(Pointer(Data16c), Rgn);

 // read a block of data into array of array of Integer

 Fit.DataRep := rep32c;

 Fit.DataPrepare(Pointer(Data32c), Rgn);

 Fit.DataRead(Pointer(Data32c), Rgn);

 // ... some code

 // free

 Fit.Stream.Free;

 Fit.Free;

 Data64f := nil;

 Data16c := nil;

 Data32c := nil;

end;
