An example of using the constants cCastChars


1. We will read from the header a value of line "as is" without conversion in a variable of type String


Input



10

11

12

header fragment

TIME-OBS= '19:29:12.401'       / exposure end time                              

RA      = '          16 13 35' / Right Ascension of image center                

OBSERVER= '    M. d''Artagnan' / observer name                                  


var

 Fit: TFitsFileFrame;

 Index: Integer;

 Item: TLineItem;

 V1, V2, V3: string;

 Res: string;

begin

 // open file for read

 Fit := TFitsFileFrame.CreateJoin('some.fit', cFileRead);

 // search and read for line with a keyword 'TIME-OBS'

 Index := Fit.LineBuilder.IndexOf('TIME-OBS');

 if Index >= 0 then

 begin

   Item := Fit.LineBuilder.Items[Index, cCastChars];

   V1   := Item.Value.Str;

 end;

 // search and read for line with a keyword 'RA'

 Index := Fit.LineBuilder.IndexOf('RA');

 if Index >= 0 then

 begin

   Item := Fit.LineBuilder.Items[Index, cCastChars];

   V2   := Item.Value.Str;

 end;

 // search and read for line with a keyword 'OBSERVER'

 Index := Fit.LineBuilder.IndexOf('OBSERVER');

 if Index >= 0 then

 begin

   Item := Fit.LineBuilder.Items[Index, cCastChars];

   V3   := Item.Value.Str;

 end;

 // output

 Res := Format('V1 = %s'#13'V2 = %s'#13'V3 = %s', [V1, V2, V3]);

 ShowMessage(Res);

 // free

 Fit.Free;

end;


Output


V1 = '19:29:12.401'

V2 = '          16 13 35'

V3 = '    M. d''Artagnan'
