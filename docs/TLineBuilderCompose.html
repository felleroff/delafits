function Compose(Item: TLineItem; Cast: Integer): string;


Description:


The method converts Item into header line. Cast defines a rule of build a the line of header.


Number of characters in the header line ~80


Code Sample:


var

 Fit: TFitsHeader;

 Item: TLineItem;

 Line: string;

begin

 // open file for read

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileRead));

 // build item

 Item.Keyword       := 'DATE-OBS';

 Item.ValueIndicate := True;

 Item.Value.Dtm     := Now;

 Item.NoteIndicate  := True;

 Item.Note          := 'my date only';

 // compose line

 Line := Fit.LineBuilder.Compose(Item, cCastDate);

 ShowMessage(Line + #13 + IntToStr(Length(Line)));

 // free

 Fit.Stream.Free;

 Fit.Free;

end;


Output
