Unit: DeLaFitsCommon


Type: function


function BitPixByteSize(Value: TBitPix): Byte;


Description:


The function returns the size of a variable TBitPix (bit-depth of data) in bytes.

For the value biUnknown function returns 0.


Code Sample:


var

 R64f, R32f: Byte;

 R08u, R16c, R32c, R64c: Byte;

 Rxxx: Byte;

begin

 R64f := BitPixByteSize(bi64f);

 R32f := BitPixByteSize(bi32f);

 R08u := BitPixByteSize(bi08u);

 R16c := BitPixByteSize(bi16c);

 R32c := BitPixByteSize(bi32c);

 R64c := BitPixByteSize(bi64c);

 Rxxx := BitPixByteSize(biUnknown);

end;


Output


R64f = 8

R32f = 4

R08u = 1

R16c = 2

R32c = 4

R64c = 8

Rxxx = 0
