Unit: DeLaFitsString


Type: function


function BitPixToStri(const Value: TBitPix): string; overload;


function BitPixToStri(const Value: Integer): string; overload;


Description:


The function converts TBitPix into String.

The function is overloaded: the Value can be specified as TBitPix or as Integer.

Value (as Integer) must be equal to the standard value: -64, -32, 08, 16, 32, 64.

If Value is not standard value, the function returns the text representation of values Value.


Code Sample:


var

 S1, S2: string;

 S3, S4: string;

 S5, S5, S7: string;

begin

 // Value: TBitPix

 S1 := BitPixToStri(bi64f);

 S2 := BitPixToStri(bi16c);

 // Value: Integer

 S3 := BitPixToStri(-64);

 S4 := BitPixToStri(16);

 // Invalid Value

 S5 := BitPixToStri(biUnknown);

 S6 := BitPixToStri(0);

 S7 := BitPixToStri(-55);

end;


Output


// Value: TBitPix


S1 = '-64'

S2 = '16'


// Value: Integer


S3 = '-64'

S4 = '16'


// Invalid Value


S5 = '0'

S6 = '0'

S7 = '-55'
