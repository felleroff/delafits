procedure Read(Index: Integer; out Item: TLineItem; Cast: Integer);


Description:


The method reads the header line with the index Index.

Line - is a record from of items <keyword-separator-value-separator-note>, see TLineItem.


Numbering of lines a header begins with zero


Code Sample:


Input



09

header fragment

TIME-OBS= '19:29:12.401'       / exposure end time                              


var

 Fit: TFitsHeader;

 Line: string;

 Item1: TLineItem;

 Item2: TLineItem;

begin

 // open file for read

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileRead));

 // read header line as string and TTime

 Fit.LineBuilder.Read(9, Item1, cCastString);

 Fit.LineBuilder.Read(9, Item2, cCastTime);

 // free

 Fit.Stream.Free;

 Fit.Free;

end;


Output


// Value as string


Item1.Keyword       = 'TIME-OBS';

Item1.ValueIndicate = True;

Item1.Value.Str     = '19:29:12.401'

Item1.NoteIndicate  = True;

Item1.Note          = 'exposure end time';


// Value as TTime


Item2.Keyword       = 'TIME-OBS';

Item2.ValueIndicate = True;

Item2.Value.Dtm     = 30.12.1899 19:29:12.401

Item2.NoteIndicate  = True;

Item2.Note          = 'exposure end time';
