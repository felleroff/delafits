Functions ››

Unit: DeLaFitsCommon


Type: function


function IntToBitPix(Value: Integer): TBitPix;


Description:


The function converts Integer into TBitPix. Value has to be equal to one of values:-64,-32, 8, 16, 32, 64, in another case the function returns the value biUnknown.


Code Sample:


var

 R64f, R32f: TBitBix;

 R08u, R16c, R32c, R64c: TBitBix;

 Rxx1, Rxx2: TBitBix;

begin

 R64f := IntToBitPix(-64);

 R32f := IntToBitPix(-32);

 R08u := IntToBitPix(8);

 R16c := IntToBitPix(16);

 R32c := IntToBitPix(32);

 R64c := IntToBitPix(64);

 Rxx1 := IntToBitPix(0);

 Rxx2 := IntToBitPix(123);

end;


Output


R64f = bi64f

R32f = bi32f

R08u = bi08u

R16c = bi16c

R32c = bi32c

R64c = bi64c

Rxx1 = biUnknown

Rxx2 = biUnknown
