function LineIndexOf(Keyword: string; StartIndex: Integer = 0): Integer;


Description:


Method returns the line index with the keyword Keyword. Linear search begins with a line number StartIndex, case insensitive.


Method returns -1 if the line is not found.


Numbering of lines a header begins with zero.


Code Sample:


Input



09

header fragment

TIME-OBS= '19:29:12.401'       / exposure end time                              


var

 Fit: TFitsHeader;

 Index1, Index2: Integer;

begin

 Fit := TFitsHeader.CreateJoin(TFileStream.Create('some.fit', cFileRead));

 Index1 := Fit.LineIndexOf('Time-Obs');

 Index2 := Fit.LineIndexOf('Absurd');

 Fit.Stream.Free;

 Fit.Free;

end;


Output


Index1 = 9

Index2 = -1
